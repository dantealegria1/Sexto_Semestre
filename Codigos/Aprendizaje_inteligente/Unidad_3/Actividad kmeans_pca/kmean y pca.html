paises = {}

def convertir_decimal(coordenada):
    # Reemplazar caracteres especiales por espacio para facilitar el análisis
    coordenada = re.sub(r'[°′″]', ' ', coordenada)
    partes = coordenada.split()

    if len(partes) == 4:  # Grados, Minutos, Segundos y Dirección
        grados, minutos, segundos, direccion = partes
    elif len(partes) == 3:  # Grados, Minutos y Dirección
        grados, minutos, direccion = partes
        segundos = '0'
    elif len(partes) == 2:  # Grados y Dirección
        grados, direccion = partes
        minutos = '0'
        segundos = '0'
    else:
        raise ValueError(f"Formato de coordenada no válido: {coordenada}")

    # Convertir a valores decimales
    decimal = float(grados) + float(minutos) / 60 + float(segundos) / 3600

    # Aplicar signo negativo para S y W
    if direccion in 'SW':
        decimal = -decimal

    return decimal

def bar_plot(centros, labels, cluster=None, var=None):
  from math import ceil, floor
  from seaborn import color_palette
  colores = color_palette()
  minimo = floor(centros.min()) if floor(centros.min()) < 0 else 0
  def inside_plot(valores,labels,titulo):
    plt.barh(range(len(valores)), valores, 1/1.5, color=colores)
    plt.xlim(minimo, ceil(centros.max()))
    plt.title(titulo)
  if var is not None:
    centros = np.array(n[[x in var for x in labels]] for n in centros)
    colores = [colores[x % len(colores)] for x, i in enumerate(labels) if i in var]
    labels = labels[[x in var for x in labels]]
  if cluster is None:
    for i in range(len(centros)):
      plt.subplot(1,centros.shape[0],i+1)
      inside_plot(centros[i].tolist(),labels,"Cluster "+str(i))
      plt.yticks(range(len(labels)), labels) if i == 0 else plt.yticks([])
  else:
    pos = 1
    for i in cluster:
      plt.subplot(1,len(cluster),pos)
      inside_plot(centros[i].tolist(),labels,"Cluster "+str(i))
      plt.yticks(range(len(labels)), labels) if pos == 1 else plt.yticks([])
      pos += 1
  plt.show()
plt.figure(1, figsize=(12, 8))
